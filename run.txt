#!/bin/bash
set -e

echo "[COLAB] Preparing Alien-LLM environment..."

# Install dependencies
apt-get update -y
apt-get install -y libeigen3-dev cmake build-essential

# Install python dependencies
pip install -r requirements.txt

# Download datasets
python3 download_datasets.py

# Build the project (Clear existing build to avoid path mismatches)
rm -rf build
mkdir -p build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
make -j$(nproc)


# Run the model
echo "[COLAB] Starting Training (BLITZ MODE)..."
cd ..

./build/alien_llm
