#!/bin/bash
set -e

echo "[COLAB] Preparing Alien-LLM environment..."

# Install dependencies
apt-get update -y
apt-get install -y libeigen3-dev cmake build-essential

# Install python dependencies
pip install -r requirements.txt

# Download datasets
python3 download_datasets.py

# Build the project (Clear existing build to avoid path mismatches)
rm -rf build
mkdir -p build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
make -j$(nproc)


# Run the model
echo "[COLAB] Starting Training (QUICK MODE)..."
cd ..

# RUN OPTIONS:
# For full training: ./build/alien_llm
# For resume: ./build/alien_llm --resume
# For quick test: ./build/alien_llm --quick

./build/alien_llm --quick
